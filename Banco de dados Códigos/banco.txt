create table jogador(
	telefone varchar(14) primary key,
	nome varchar(30)		
)

create table usuario(
	id serial primary key,
	nome varchar(30),
	username varchar(30) UNIQUE,
	senha varchar(10),
	admin boolean
	
)


create table concurso(

	id integer primary key,
	dataInicio date not null,
	horaInicio time [4] not null,
	dataFim date not null,
	horaFim time [4] not null,
	dezena integer not null,
	foreign key (dezena) references dezena(id)
	
)


create table aposta(
	id serial primary key,
	concurso integer not null,
	jogador varchar(14) not null,
	usuario integer not null,
	data timestamp not null,
	preco money,
	dezena integer not null,
	foreign key (dezena) references dezena(id),
	foreign key (concurso) references concurso(id),
	foreign key (jogador) references jogador(telefone),
	foreign key (usuario) references usuario(id)
	
)

create table dezena(
	id integer primary key,
	d1 smallint,
	d2 smallint,
	d3 smallint,
	d4 smallint,
	d5 smallint,
	d6 smallint,
	d7 smallint,
	d8 smallint,
	d9 smallint,
	d10 smallint
	
	
)
